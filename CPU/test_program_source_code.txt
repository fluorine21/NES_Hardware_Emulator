
//Simple program for adding numbers in memory in zero page
//Adds 5 numbers from addr 0 - 4 and stores result at 5

LDX #0
TXA 
CLC
ADC $00,X
INX
ADC $00,X
INX
ADC $00,X
INX
ADC $00,X
INX
ADC $00,X
INX
STA $00,X


//Add with branch
LDX #0
TXA
CLC
//Load the current variable from memory
L1 CLC 
ADC *$00,X
//Compare X to 5
CPX #5
BMI L1
STA $00,x
